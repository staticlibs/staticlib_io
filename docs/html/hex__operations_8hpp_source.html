96 inline std::string format_hex_and_plain(const std::string& hexstr, const std::string& plain) {
97  if (hexstr.empty() || plain.empty()) return std::string();
98 
99  // prettify hex
100  auto pretty = std::string();
101  for (size_t i = 0; i < hexstr.length() - 1; i += 2 ) {
102  if (pretty.length() > 0) {
103  pretty.push_back(' ');
104  }
105  pretty.push_back(hexstr[i]);
106  pretty.push_back(hexstr[i + 1]);
107  }
108 
109  // cleanup plain
110  auto clean = std::string();
111  for (size_t i = 0; i < plain.length(); i++) {
112  char ch = plain[i];
113  if (ch < 32 || std::isspace(ch)) {
114  clean.push_back(' ');
115  } else {
116  clean.push_back(ch);
117  }
118  }
119 
120  // return formatted
121  return pretty + " [" + clean + "]";
122 }
