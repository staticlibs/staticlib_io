134  std::streamsize read(span<char> span) {
135  size_t ulen = span.size();
136  // return from buffer
137  if (ulen <= avail) {
138  std::memcpy(span.data(), buffer.data() + pos, ulen);
139  pos += ulen;
140  avail -= ulen;
141  return span.size_signed();
142  }
143  // copy all data already available
144  size_t uhead = avail;
145  if (!sl::support::is_streamsize(uhead)) {
146  uhead = static_cast<size_t> (std::numeric_limits<std::streamsize>::max());
147  }
148  if (uhead > 0) {
149  std::memcpy(span.data(), buffer.data() + pos, uhead);
150  }
151  pos = 0;
152  avail = 0;
153  std::streamsize head = static_cast<std::streamsize> (uhead);
154  // try to guess whether to do direct read, or fill buffer first
155  if (ulen > buffer.size()) {
156  // read directly into the destination
157  size_t result = read_into_buffer(span.data(), uhead, ulen - uhead);
158  size_t out = result + uhead;
159  return out > 0 ? out : std::char_traits<char>::eof();
160  }
161  // fill buffer
162  avail = read_into_buffer(buffer.data(), 0, buffer.size());
163  if (avail > 0) {
164  // copy tail from buffer
165  size_t to_read_req = ulen - uhead;
166  size_t to_read = to_read_req <= avail ? to_read_req : avail;
167  if (!sl::support::is_streamsize(to_read + uhead)) {
168  to_read = static_cast<size_t> (std::numeric_limits<std::streamsize>::max()) - uhead;
169  }
170  std::memcpy(span.data() + uhead, buffer.data(), to_read);
171  pos = to_read;
172  avail -= to_read;
173  // note that head >= 0, and to_read > 0
174  return static_cast<std::streamsize>(to_read + uhead);
175  }
176  return head == 0 ? std::char_traits<char>::eof() : head;
177  }

188  std::string read_line(const std::string& ending = "\n", size_t max_length = (1<<16)) {
189  std::string line;
190  char ch;
191  while(line.length() < max_length && 1 == read({std::addressof(ch), 1})) {
192  line.push_back(ch);
193  if (ending.length() > 0 && ch == ending.back() && line.length() >= ending.length()) {
194  bool matched = true;
195  size_t sdiff = line.length() - ending.length();
196  for (int i = static_cast<int>(ending.length()) - 2; i >= 0; i--) {
197  if (ending[i] != line[sdiff + i]) {
198  matched = false;
199  break;
200  }
201  }
202  if (matched) {
203  for (size_t i = 0; i < ending.length(); i++) {
204  line.pop_back();
205  }
206  if (line.length() > 0) {
207  break;
208  }
209  }
210  }
211  }
212  return line;
213  }

233 private:
234  // repeatable source read logic
235  size_t read_into_buffer(char* buf, size_t offset, size_t length) {
236  if (!exhausted) {
237  size_t result = 0;
238  while (result < length) {
239  size_t ulen = length - result;
240  if (!sl::support::is_streamsize(ulen)) {
241  ulen = static_cast<size_t> (std::numeric_limits<std::streamsize>::max());
242  }
243  std::streamsize amt = src.read({buf + offset + result, static_cast<std::streamsize> (ulen)});
244  if (std::char_traits<char>::eof() != amt) {
245  if (!sl::support::is_sizet(amt)) throw io_exception(TRACEMSG(
246  "Invalid result returned by underlying 'read' operation: [" + sl::support::to_string(amt) + "]"));
247  result += static_cast<size_t> (amt);
248  } else {
249  exhausted = true;
250  break;
251  }
252  }
253  return result;
254  }
255  return 0;
256  }
