136  std::streamsize write(span<const char> span) {
137  size_t ulen = span.size();
138  if (ulen >= buffer.size()) {

143  } else if (ulen <= avail) {

145  pos += ulen;
146  if (avail > ulen) {

199  void write_to_sink(const char* buf, size_t length) {
200  size_t result = 0;
201  while (result < length) {
202  size_t ulen = length - result;
203  if (!sl::support::is_streamsize(ulen)) {
204  ulen = static_cast<size_t> (std::numeric_limits<std::streamsize>::max());
205  }
206  std::streamsize amt = sink.write({buf + result, static_cast<std::streamsize>(ulen)});
207  if (!sl::support::is_sizet(amt)) throw io_exception(TRACEMSG(
208  "Invalid result returned by underlying 'write' operation: [" + sl::support::to_string(amt) + "]"));
209  result += static_cast<size_t> (amt);
