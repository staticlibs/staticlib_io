193  std::streamsize read(span<char> span) {
194  size_t ulen = span.size();
195  // return from buffer
196  std::streamsize read_buffered = copy_buffered(span.data(), ulen);
197  if (read_buffered > 0) {
198  return read_buffered;
199  }
200  // fill buffer
201  if (!exhausted) {
202  buffer.resize(0);
203  pos = 0;
204  size_t limitlen = ulen <= buf_len_limit ? ulen : buf_len_limit;
205  for(;;) {
206  char cur;
207  std::streamsize read = 0;
208  while (0 == (read = src.read({std::addressof(cur), 1})));
209  exhausted = std::char_traits<char>::eof() == read;
210  if (exhausted) {
211  break;
212  }
213  switch (state) {
214  case State::PREFIX:
215  do_prefix(cur);
216  break;
217  case State::PLACEHOLDER:
218  do_placeholder(cur);
219  break;
220  case State::MOVED_FROM:
221  throw io_exception(TRACEMSG("Invalid attempt to read from source in 'MOVED_FROM' state"));
222  }
223  if (buffer.size() >= limitlen) {
224  break;
225  }
226  }
227  }
228  // return if any
229  std::streamsize read_prepared = copy_buffered(span.data(), ulen);
230  if (read_prepared > 0) {
231  return read_prepared;
232  }
233  // return eof
234  if (placeholder.length() > 0) {
235  std::string msg = "Invalid unclosed placeholder: [" + placeholder + "]";
236  on_error(msg);
237  }
238  return std::char_traits<char>::eof();
239  }

261  void do_prefix(char cur) {
262  if (cur != prefix[ind_prefix]) {
263  if (ind_prefix > 0) {
264  size_t wp = buffer.size();
265  buffer.resize(wp + ind_prefix);
266  std::memcpy(buffer.data() + wp, prefix.c_str(), ind_prefix);
267  ind_prefix = 0;
268  }
269  buffer.push_back(cur);
270  } else {
271  ind_prefix += 1;
272  if (prefix.length() == ind_prefix) {
273  ind_prefix = 0;
274  state = State::PLACEHOLDER;
275  }
276  }
277  }

279  void do_placeholder(char cur) {
280  placeholder.push_back(cur);
281  if (placeholder.size() == max_placeholder_len) {
282  std::string msg = "Parameter name: [" + placeholder + "] is too long";
283  on_error(msg);
284  }
285  if (cur == postfix[ind_postfix]) {
286  ind_postfix += 1;
287  if (postfix.length() == ind_postfix) {
288  ind_postfix = 0;
289  placeholder.resize(placeholder.length() - postfix.length());
290  auto par = values.find(placeholder);
291  if (values.end() != par) {
292  size_t wp = buffer.size();
293  buffer.resize(wp + par->second.length());
294  std::memcpy(buffer.data() + wp, par->second.c_str(), par->second.length());
295  placeholder.clear();
296  state = State::PREFIX;
297  } else {
298  std::string msg = "Parameter: [" + placeholder + "] not found";
299  on_error(msg);
300  }
301  }
302  } else {
303  ind_postfix = 0;
304  }
305  }

307  std::streamsize copy_buffered(char* buf, size_t ulen) {
308  size_t avail = buffer.size() - pos;
309  if (avail > 0) {
310  size_t ucplen = avail <= ulen ? avail : ulen;
311  std::memcpy(buf, buffer.data() + pos, ucplen);
312  pos += ucplen;
313  return static_cast<std::streamsize>(ucplen);
314  }
315  return 0;
316  }
