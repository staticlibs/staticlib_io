84  template<typename IntType>
85  span(T* buffer, IntType length) :
86  first_ptr(buffer),
87  last_ptr(buffer) {
88  if (nullptr != buffer) {
89  if (sl::support::is_sizet(length) && sl::support::is_streamsize(length)) {
90  last_ptr += static_cast<size_t> (length);
91  } else {
92  throw bad_span_access_exception(std::string() + "Invalid 'length' span parameter specified," +
93  " length: [" + sl::support::to_string(length) + "]");
94  }
95  } else {
96  throw bad_span_access_exception(std::string() + "Invalid 'null' data pointer specified to span," +
97  " length: [" + sl::support::to_string(length) + "]");
98  }
99  }
