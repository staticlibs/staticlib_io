173  virtual int uflow() override {
174  if (exhausted) {
175  return std::char_traits<char>::eof();
176  } else {
177  throw std::ios_base::failure("'uflow' operation is not supported by unbuffered streambuf");
178  }
179  }
180 
186  Source& get_source() {
187  return source;
188  }
189 
190 protected:
198  virtual std::streamsize xsgetn(char* s, std::streamsize count) override {
199  if (!exhausted) {
200  auto res = source.read({s, count});
201  if (source_eof != res) {
202  return res;
203  } else {
204  exhausted = true;
205  return 0;
206  }
207  } else {
208  return 0;
209  }
210  }
